const tools = {
  debounce: (fn, wait = 500) => {
    let timer = null;
    return function () {
      if (timer) {
        clearTimeout(timer);
      }
      timer = setTimeout(() => {
        fn.apply(this, arguments);
      }, wait);
    };
  },
  throttle: (fn, wait = 500) => {
    let canrun = true;
    return function () {
      if (!canrun) {
        return;
      }
      canrun = false;
      setTimeout(() => {
        fn.apply(this, arguments);
        canrun = true;
      }, wait);
    };
  },
};
